# ESP32 WHEELCHAIR FIRMWARE

```cpp
#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <Preferences.h>
#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>


#define FIRMWARE_VERSION "1.0"


// MOTOR PINS

#define LEFT_FORWARD 26
#define LEFT_BACKWARD 27

#define RIGHT_FORWARD 25
#define RIGHT_BACKWARD 33


#define BUZZER 13


// WIFI

String ssid="";
String password="";

Preferences preferences;


// MQTT

const char* mqtt_server="broker.hivemq.com";


String CMD_TOPIC=
"iot/wheelchair/action";


String STATUS_TOPIC=
"iot/wheelchair/status";


String ACK_TOPIC=
"iot/wheelchair/ack";



WiFiClient espClient;

PubSubClient mqtt(espClient);



String lastCommand="";



void stopMotor()
{

digitalWrite(LEFT_FORWARD,LOW);
digitalWrite(LEFT_BACKWARD,LOW);

digitalWrite(RIGHT_FORWARD,LOW);
digitalWrite(RIGHT_BACKWARD,LOW);

}



void forwardMove()
{

digitalWrite(LEFT_FORWARD,HIGH);
digitalWrite(RIGHT_FORWARD,HIGH);


digitalWrite(LEFT_BACKWARD,LOW);
digitalWrite(RIGHT_BACKWARD,LOW);

}



void backwardMove()
{

digitalWrite(LEFT_BACKWARD,HIGH);
digitalWrite(RIGHT_BACKWARD,HIGH);


digitalWrite(LEFT_FORWARD,LOW);
digitalWrite(RIGHT_FORWARD,LOW);

}




void leftMove()
{

digitalWrite(LEFT_BACKWARD,HIGH);

digitalWrite(RIGHT_FORWARD,HIGH);


}




void rightMove()
{

digitalWrite(LEFT_FORWARD,HIGH);

digitalWrite(RIGHT_BACKWARD,HIGH);

}



void sendACK(String cmd,String id)
{


StaticJsonDocument<200> doc;


doc["device"]="ESP32_WHEELCHAIR";

doc["command"]=cmd;

doc["command_id"]=id;

doc["result"]="SUCCESS";


char buffer[200];


serializeJson(doc,buffer);



mqtt.publish(
ACK_TOPIC.c_str(),
buffer
);


}




// MQTT CALLBACK


void callback(char* topic,
byte* payload,
unsigned int length)
{


String msg="";


for(int i=0;i<length;i++)
{

msg+=(char)payload[i];

}



StaticJsonDocument<300> doc;


deserializeJson(doc,msg);



String command=
doc["command"];


String id=
doc["command_id"];



if(id==lastCommand)
return;


lastCommand=id;



if(command=="FORWARD")
{
forwardMove();
}


else if(command=="BACKWARD")
{
backwardMove();
}


else if(command=="LEFT")
{
leftMove();
}


else if(command=="RIGHT")
{
rightMove();
}



else if(command=="STOP")
{
stopMotor();
}



else if(command=="HORN")
{

digitalWrite(BUZZER,HIGH);

delay(300);

digitalWrite(BUZZER,LOW);

}



sendACK(command,id);



}







void connectMQTT()
{


while(!mqtt.connected())
{


String id="CHAIR_";

id+=random(1000);



if(mqtt.connect(id.c_str()))
{

mqtt.subscribe(
CMD_TOPIC.c_str()
);


}

else

delay(2000);


}


}




void setupWiFi()
{


WiFi.begin(
ssid.c_str(),
password.c_str()
);



while(WiFi.status()!=WL_CONNECTED)
{

delay(500);

}



}





void loadWiFi()
{


preferences.begin("wifi",true);


ssid=
preferences.getString("ssid","");


password=
preferences.getString("pass","");


preferences.end();


}





void setup()
{


Serial.begin(115200);



pinMode(LEFT_FORWARD,OUTPUT);
pinMode(LEFT_BACKWARD,OUTPUT);

pinMode(RIGHT_FORWARD,OUTPUT);
pinMode(RIGHT_BACKWARD,OUTPUT);


pinMode(BUZZER,OUTPUT);


stopMotor();



loadWiFi();



setupWiFi();



mqtt.setServer(
mqtt_server,
1883
);


mqtt.setCallback(callback);



}






unsigned long timer=0;



void loop()
{


if(!mqtt.connected())
connectMQTT();


mqtt.loop();




if(millis()-timer>3000)
{

timer=millis();


StaticJsonDocument<200> doc;



doc["device"]="ESP32_WHEELCHAIR";

doc["status"]="ONLINE";

doc["wifi"]=WiFi.localIP().toString();



char data[200];


serializeJson(doc,data);



mqtt.publish(
STATUS_TOPIC.c_str(),
data
);



}



}
```
